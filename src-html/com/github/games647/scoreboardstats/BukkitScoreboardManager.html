018/**
019 * Managing the scoreboard access.
020 *
021 * @see com.github.games647.scoreboardstats.protocol.PacketSbManager
022 */
023public class BukkitScoreboardManager extends SbManager {
024
025    private static final String CRITERIA = "dummy";
026
027    private final boolean oldBukkit;
028
029    /**
030     * Initialize scoreboard manager.
031     *
032     * @param pluginInstance the ScoreboardStats instance
033     */
034    public BukkitScoreboardManager(ScoreboardStats pluginInstance) {
035        super(pluginInstance);
036
037        oldBukkit = isOldBukkit();
038    }
039
040    @Override
041    public void createScoreboard(Player player) {
042        final Objective oldObjective = player.getScoreboard().getObjective(DisplaySlot.SIDEBAR);
043        if (!isValid(player) || oldObjective != null && !TEMP_SB_NAME.equals(oldObjective.getName())) {
044            //Check if another scoreboard is showing
045            return;
046        }
047
048        //Creates a new scoreboard and a new objective
049        final Scoreboard board = Bukkit.getScoreboardManager().getNewScoreboard();
050        final Objective objective = board.registerNewObjective(SB_NAME, CRITERIA);
051        objective.setDisplaySlot(DisplaySlot.SIDEBAR);
052        objective.setDisplayName(Settings.getTitle());
053
054        try {
055            player.setScoreboard(board);
056        } catch (IllegalStateException stateEx) {
057            //the player logged out - fail silently
058            return;
059        }
060
061        sendUpdate(player, true);
062        //Schedule the next tempscoreboard show
063        scheduleShowTask(player, true);
064    }
065
066    @Override
067    public void unregister(Player player) {
068        if (player.isOnline()) {
069            for (Objective objective : player.getScoreboard().getObjectives()) {
070                final String objectiveName = objective.getName();
071                if (objectiveName.startsWith(SB_NAME)) {
072                    objective.unregister();
073                }
074            }
075        }
076    }
077
078    @Override
079    public void sendUpdate(Player player) {
080        final Objective objective = player.getScoreboard().getObjective(DisplaySlot.SIDEBAR);
081        if (objective == null) {
082            //The player has no scoreboard so create one
083            createScoreboard(player);
084        } else {
085            sendUpdate(player, false);
086        }
087    }
088
089    @Override
090    public void createTopListScoreboard(Player player) {
091        final Objective oldObjective = player.getScoreboard().getObjective(DisplaySlot.SIDEBAR);
092        if (!isValid(player) || oldObjective == null
093                || !oldObjective.getName().startsWith(SB_NAME)) {
094            //Check if another scoreboard is showing
095            return;
096        }
097
098        //remove old scores
099        if (TEMP_SB_NAME.equals(oldObjective.getName())) {
100            oldObjective.unregister();
101        }
102
103        final Scoreboard board = player.getScoreboard();
104        final Objective objective = board.registerNewObjective(TEMP_SB_NAME, CRITERIA);
105        objective.setDisplaySlot(DisplaySlot.SIDEBAR);
106        objective.setDisplayName(Settings.getTempTitle());
107
108        try {
109            player.setScoreboard(board);
110        } catch (IllegalStateException stateEx) {
111            //the player logged out - fail silently
112            return;
113        }
114
115        //Colorize and send all elements
116        for (Map.Entry<String, Integer> entry : plugin.getStatsDatabase().getTop()) {
117            final String scoreName = stripLength(Settings.getTempColor() + entry.getKey());
118            sendScore(objective, scoreName, entry.getValue(), true);
119        }
120
121        //schedule the next normal scoreboard show
122        scheduleShowTask(player, false);
123    }
124
125    @Override
126    public void update(Player player, String itemName, int newScore) {
127        final Scoreboard scoreboard = player.getScoreboard();
128        if (scoreboard != null) {
129            final Objective objective = scoreboard.getObjective(SB_NAME);
130            if (objective != null) {
131                sendScore(objective, itemName, newScore, false);
132            }
133        }
134    }
135
136    @Override
137    protected void sendUpdate(Player player, boolean complete) {
138        final Objective objective = player.getScoreboard().getObjective(DisplaySlot.SIDEBAR);
139        //don't override other scoreboards
140        if (SB_NAME.equals(objective.getName())) {
141            final Iterator<Map.Entry<String, String>> iter = Settings.getItems();
142            while (iter.hasNext()) {
143                final Map.Entry<String, String> entry = iter.next();
144                final String title = entry.getKey();
145                final String variable = entry.getValue();
146
147                try {
148                    final ReplaceEvent replaceEvent = replaceManager.getScore(player, variable, title, 0, complete);
149                    if (replaceEvent.isModified()) {
150                        sendScore(objective, title, replaceEvent.getScore(), complete);
151                    }
152                } catch (UnknownVariableException ex) {
153                    //Remove the variable becaue we can't replace it
154                    iter.remove();
155                    plugin.getLogger().info(Lang.get("unknownVariable", variable));
156                }
157            }
158        }
159    }
160
161    private void sendScore(Objective objective, String title, int value, boolean complete) {
162        final String scoreName = expandScore(title, objective);
163
164        Score score;
165        if (oldBukkit) {
166            //Bukkit.getOfflinePlayer performance work around
167            score = objective.getScore(new FastOfflinePlayer(scoreName));
168        } else {
169            score = objective.getScore(scoreName);
170        }
171
172        if (complete && value == 0) {
173            /*
174             * Workaround because the value from Bukkit is set as default to zero and Bukkit sends only
175             * the packet if the value changes so we have to change it to another value earlier
176             */
177            score.setScore(1337);
178        }
179
180        score.setScore(value);
181    }
182
183    private String expandScore(String scoreName, Objective objective) {
184        String cleanScoreName = scoreName;
185        final int titleLength = cleanScoreName.length();
186        if (titleLength > 16) {
187            final Scoreboard scoreboard = objective.getScoreboard();
188            //Could maybe cause conflicts but .substring could also make conflicts if they have the same beginning
189            final String teamId = String.valueOf(cleanScoreName.hashCode());
190
191            Team team = scoreboard.getTeam(teamId);
192            if (team == null) {
193                team = scoreboard.registerNewTeam(teamId);
194                team.setPrefix(cleanScoreName.substring(0, 16));
195                if (titleLength > 32) {
196                    //we already validated that this one can only be 48 characters long in the Settings class
197                    team.setSuffix(cleanScoreName.substring(32));
198                    cleanScoreName = cleanScoreName.substring(16, 32);
199                } else {
200                    cleanScoreName = cleanScoreName.substring(16);
201                }
202
203                team.setDisplayName(cleanScoreName);
204                //Bukkit.getOfflinePlayer performance work around
205                team.addPlayer(new FastOfflinePlayer(cleanScoreName));
206            } else {
207                cleanScoreName = team.getDisplayName();
208            }
209        }
210
211        return cleanScoreName;
212    }
213
214    private boolean isOldBukkit() {
215        final int compare = Version.compare("1.7.8", Version.getMinecraftVersionString());
216        if (compare >= 0) {
217            try {
218                Objective.class.getDeclaredMethod("getScore", String.class);
219                //We have access to the new method
220                return false;
221            } catch (NoSuchMethodException noSuchMethodEx) {
222                //since we have an extra class for it (FastOfflinePlayer)
223                //we can fail silently
224            }
225        }
226
227        //The version is under 1.7.8 so the method doesn't exist
228        return true;
229    }
