001package com.github.games647.scoreboardstats;
002
003import java.text.MessageFormat;
004import java.util.Locale;
005import java.util.ResourceBundle;
006
007import org.bukkit.Bukkit;
008
009/**
010 * Separate the code and the messages.
011 */
012public class Lang {
013
014    private static final Lang INSTANCE = new Lang();
015
016    /**
017     * Get a localized string.
018     *
019     * @param key the localization key
020     * @return the localized string
021     */
022    public static String get(String key) {
023        //Static method wrapper
024        return INSTANCE.getFormatted(key);
025    }
026
027    /**
028     * Get a localized string.
029     *
030     * @param key the localization key
031     * @param arguments arguments for formatting strings
032     * @return localized
033     */
034    public static String get(String key, Object... arguments) {
035        //Static method wrapper
036        return INSTANCE.getFormatted(key, arguments);
037    }
038
039    private final ResourceBundle defaultMessages;
040
041    /**
042     * Initialize localization manager
043     */
044    public Lang() {
045        //Warning this can be null. - ToDo check it
046        //JavaPlugin.getPlugin isn't available in 1.5
047        final ScoreboardStats plugin = (ScoreboardStats) Bukkit.getPluginManager().getPlugin("ScoreboardStats");
048        final ClassLoader classLoader = plugin.getClassLoaderBypass();
049
050        defaultMessages = ResourceBundle.getBundle("messages", Locale.getDefault(), classLoader);
051    }
052
053    private String getFormatted(String key, Object... arguments) {
054        if (defaultMessages.containsKey(key)) {
055            String result = defaultMessages.getString(key);
056            if (arguments.length != 0) {
057                //If there are arguments use messageformat to replace
058                //proparly cache the formatting instance
059                result = MessageFormat.format(result, arguments);
060            }
061
062            return result;
063        }
064
065        //fail silently for now
066        return "";
067    }
068}
