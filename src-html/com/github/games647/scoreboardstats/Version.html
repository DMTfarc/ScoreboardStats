001package com.github.games647.scoreboardstats;
002
003import com.google.common.base.Objects;
004import com.google.common.collect.ComparisonChain;
005
006import java.util.regex.Matcher;
007import java.util.regex.Pattern;
008
009import org.apache.commons.lang.builder.ToStringBuilder;
010import org.bukkit.Bukkit;
011
012/**
013 * Version class for comparing and detecting Minecraft and other versions
014 */
015public class Version implements Comparable<Version> {
016
017    //thanks to the author of ProtocolLib aadnk
018    private static final String VERSION_REGEX = ".*\\(.*MC.\\s*([a-zA-z0-9\\-\\.]+)\\s*\\)";
019
020    private static boolean UUID_COMPATIBLE;
021
022    static {
023        try {
024            UUID_COMPATIBLE = compare("1.7", getMinecraftVersionString()) >= 0;
025        } catch (Exception ex) {
026            UUID_COMPATIBLE = false;
027        }
028    }
029
030    /**
031     * Gets whether server supports UUIDs
032     *
033     * @return whether server supports UUIDs
034     */
035    public static boolean isUUIDCompatible() {
036        return UUID_COMPATIBLE;
037    }
038
039    /**
040     * Gets the Minecraft version
041     *
042     * @return the Minecraft version
043     */
044    public static Version getMinecraftVersion() {
045        return new Version(getMinecraftVersionString());
046    }
047
048    /**
049     * Gets the Minecraft version as string
050     *
051     * @return the Minecraft version as string
052     */
053    public static String getMinecraftVersionString() {
054        return getVersionStringFromServer(Bukkit.getVersion());
055    }
056
057    /**
058     * Compares the version with checking the first three numbers
059     *
060     * @param expected the object to be compared.
061     * @param version the object to be compared.
062     * @return 1 version is higher; 0 both are equal; -1 version is lower<br>
063     *          a negative integer, zero, or a positive integer as this object
064     *          is less than, equal to, or greater than the specified object.
065     */
066    public static int compare(String expected, String version) {
067        final int[] expectedParts = parse(version);
068        final int[] versionParts = parse(expected);
069
070        return ComparisonChain.start()
071                .compare(expectedParts[0], versionParts[0])
072                .compare(expectedParts[1], versionParts[1])
073                .compare(expectedParts[2], versionParts[2])
074                .result();
075    }
076
077    /**
078     * Separate the version into major, minor, build integers
079     *
080     * @param version the version that should be parsed
081     * @return the version parts
082     * @throws IllegalArgumentException if the version doesn't contains only positive numbers separated by max. 5 dots.
083     */
084    public static int[] parse(String version) throws IllegalArgumentException {
085        //exludes spaces which could be added by mistake and exclude build suffixes
086        final String trimedVersion = version.trim().split("\\-")[0];
087        if (!trimedVersion.matches("\\d+(\\.\\d+){0,5}")) {
088            //check if it's a format like '1.5'
089            throw new IllegalArgumentException("Invalid format: " + version);
090        }
091
092        final int[] versionParts = new int[3];
093
094        //escape regEx and split by dots
095        final String[] split = trimedVersion.split("\\.");
096        //We check if the length has min 1 entry.
097        for (int i = 0; i < split.length && i < versionParts.length; i++) {
098            versionParts[i] = Integer.parseInt(split[i]);
099        }
100
101        return versionParts;
102    }
103
104    private static String getVersionStringFromServer(String versionString) {
105        final Pattern versionPattern = Pattern.compile(VERSION_REGEX);
106        final Matcher versionMatche = versionPattern.matcher(versionString);
107
108        if (versionMatche.matches() && versionMatche.group(1) != null) {
109            return versionMatche.group(1);
110        } else {
111            //fallback to the toString() method
112            final String[] split = Bukkit.getServer().toString().split("[,}]");
113            for (String element : split) {
114                if (element.contains("minecraftVersion=")) {
115                    return element.split("minecraftVersion=")[1];
116                }
117            }
118
119            //Couldn't extract the version
120            throw new IllegalStateException("Cannot parse version String '" + versionString);
121        }
122    }
123
124    private final int major;
125    private final int minor;
126    private final int build;
127
128    /**
129     * Creates a new version based on this string.
130     *
131     * @param version the version string
132     * @throws IllegalArgumentException if the string doesn't match a version format
133     */
134    public Version(String version) throws IllegalArgumentException {
135        final int[] versionParts = parse(version);
136
137        major = versionParts[0];
138        minor = versionParts[1];
139        build = versionParts[2];
140    }
141
142    /**
143     * Creates a new version based on these values.
144     *
145     * @param major the major version
146     * @param minor the minor version
147     * @param build the build version
148     */
149    public Version(int major, int minor, int build) {
150        this.major = major;
151        this.minor = minor;
152        this.build = build;
153    }
154
155    /**
156     * Gets the major value
157     *
158     * @return the major value
159     */
160    public int getMajor() {
161        return major;
162    }
163
164    /**
165     * Gets the minor value
166     *
167     * @return the minor value
168     */
169    public int getMinor() {
170        return minor;
171    }
172
173    /**
174     * Gets the build value
175     *
176     * @return the build value
177     */
178    public int getBuild() {
179        return build;
180    }
181
182    @Override
183    public int compareTo(Version other) {
184        return ComparisonChain.start()
185                .compare(major, other.major)
186                .compare(minor, other.minor)
187                .compare(build, other.build)
188                .result();
189    }
190
191    @Override
192    public int hashCode() {
193        return Objects.hashCode(major, minor, build);
194    }
195
196    @Override
197    public boolean equals(Object obj) {
198        //ignores also null
199        if (obj instanceof Version) {
200            final Version other = (Version) obj;
201            return major == other.major
202                    && minor == other.minor
203                    && build == other.build;
204        } else {
205            return false;
206        }
207    }
208
209    @Override
210    public String toString() {
211        return ToStringBuilder.reflectionToString(this);
212    }
213}
