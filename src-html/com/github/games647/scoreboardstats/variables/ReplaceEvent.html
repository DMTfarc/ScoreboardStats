001package com.github.games647.scoreboardstats.variables;
002
003import org.apache.commons.lang.StringUtils;
004
005/**
006 * Represents a variable replace action
007 */
008public class ReplaceEvent {
009
010    private final String variable;
011    private final boolean textVariable;
012
013    private boolean constant;
014    private boolean modified;
015//    private boolean global;
016    private String displayText;
017    private int score;
018//    private int updateInterval;
019
020    /**
021     * Creates a replace event
022     *
023     * @param variable the to replaced variable
024     * @param textVariable whether it should return an String or Integer
025     * @param constant whether it will be updated with an event handler
026     * @param displayText the scoreboard item name
027     * @param score the scoreboard item score
028     */
029    public ReplaceEvent(String variable, boolean textVariable, boolean constant, String displayText, int score) {
030        this.variable = variable;
031        this.textVariable = textVariable;
032        this.constant = constant;
033        this.displayText = displayText;
034        this.score = score;
035//        this.global = global;
036//        this.updateInterval = interval;
037    }
038
039    /**
040     * Get whether this event is modified
041     *
042     * @return whether this event is modified
043     */
044    public boolean isModified() {
045        return modified;
046    }
047
048    /**
049     * Get hether the variable is in the display name
050     *
051     * @return whether the variable is in the display name
052     */
053    public boolean isTextVariable() {
054        return textVariable;
055    }
056
057    /**
058     * Get whether it will be update with an event handler
059     *
060     * @return whether it will be update with an event handler
061     */
062    public boolean isConstant() {
063        return constant;
064    }
065
066    /**
067     * Set the constant status of the variable
068     *
069     * @param constant whether it will be update with an event handler
070     */
071    public void setConstant(boolean constant) {
072        this.constant = constant;
073    }
074
075    /**
076     * Get the display name
077     *
078     * @return the display name of this scoreboard item
079     */
080    public String getDisplayText() {
081        return displayText;
082    }
083
084    /**
085     * Set the entire display name of this scoreboard item to this text
086     *
087     * @param newDisplayText the new display name
088     */
089    public void setDisplayText(String newDisplayText) {
090//        if (!displayText.equals(newDisplayText)) {
091            touch();
092            this.displayText = newDisplayText;
093//        }
094    }
095
096    /**
097     * Get the score of this scoreboard item
098     *
099     * @return the scoreboard item score
100     */
101    public int getScore() {
102        return score;
103    }
104
105    /**
106     * Set the score of this scoreboard item
107     *
108     * @param newScore new item score
109     */
110    public void setScore(int newScore) {
111//        if (score != newScore) {
112            touch();
113            this.score = newScore;
114//        }
115    }
116
117    /**
118     * Set the score or display name of this scoreboard item
119     *
120     * @param newScore the new value
121     */
122    public void setScoreOrText(int newScore) {
123        if (textVariable) {
124            setDisplayText(StringUtils.replace(displayText, variable, Integer.toString(newScore), 1));
125        } else {
126            setScore(newScore);
127        }
128    }
129
130    /**
131     * Set the score or display name of this scoreboard item
132     *
133     * @param replacedVariable the new value
134     */
135    public void setScoreOrText(String replacedVariable) {
136        if (textVariable) {
137            //StringUtil.replace is fater because it doesn't create a pattern in the background
138            //and we need the performance as the replace method is called often
139            setDisplayText(StringUtils.replace(displayText, variable, replacedVariable, 1));
140        } else {
141            final Integer parsedInt = Integer.getInteger(variable);
142            if (parsedInt != null) {
143                setScore(parsedInt);
144            }
145        }
146    }
147
148    /**
149     * Trigger that the replacer have found the variable
150     */
151    public void touch() {
152        modified = true;
153    }
154}
