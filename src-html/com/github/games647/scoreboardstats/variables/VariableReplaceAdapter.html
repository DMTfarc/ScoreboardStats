001package com.github.games647.scoreboardstats.variables;
002
003import java.util.Arrays;
004import java.util.List;
005
006import org.apache.commons.lang.builder.ToStringBuilder;
007import org.bukkit.plugin.Plugin;
008
009/**
010 * Represents a variable replacer as an abstract class.
011 *
012 * @param <T> the type of your plugin
013 */
014public abstract class VariableReplaceAdapter<T extends Plugin> implements VariableReplacer {
015
016    private final T plugin;
017
018    private final boolean async;
019    private final boolean global;
020    private final boolean constant;
021    private final String description;
022
023    private final String[] variables;
024
025    private boolean enabled;
026
027    /**
028     * Initialize the replacer with the default values
029     *
030     * @param plugin associated plugin instance
031     * @param variables to replaced variables
032     */
033    public VariableReplaceAdapter(T plugin, String... variables) {
034        this(plugin, "&cNo description", false, false, false, variables);
035    }
036
037    /**
038     * Initialize the replacer
039     *
040     * @param global is the value the same for all players or does the replacer needs a specific player
041     * @param async is this plugin thread safe
042     * @param constant if the variable is updated based on events
043     * @param description description of all variables of this plugin
044     * @param plugin associated plugin instance
045     * @param variables to replaced variables <b>without the variable identifiers (%)</b>
046     */
047    public VariableReplaceAdapter(T plugin, String description, boolean global, boolean async, boolean constant
048            , String... variables) {
049        this.plugin = plugin;
050        this.async = async;
051        this.global = global;
052        this.constant = constant;
053
054        this.description = description;
055
056        this.variables = variables;
057    }
058
059    /**
060     * Get all variables from this replacer
061     *
062     * @return all variables which can be replaced
063     */
064    public List<String> getVariables() {
065        return Arrays.asList(variables);
066    }
067
068    /**
069     * Check whether this plugin is called async. This means that it's
070     * thread-safe.
071     *
072     * @return whether the replacer is called async
073     */
074    public boolean isAsync() {
075        return async;
076    }
077
078    /**
079     * Check whether the variables of this replacer have the same results for all
080     * players
081     *
082     * @return whether the variable values are the same for all players
083     */
084    public boolean isGlobal() {
085        return global;
086    }
087
088    /**
089     * Check whether the variables of this replacer are updated based on events
090     *
091     * @return whether the variables of this replacer are updated based on events
092     */
093    public boolean isConstant() {
094        return constant;
095    }
096
097    /**
098     * Get the default variable descriptions of all variables
099     *
100     * @return description of all variables of this plugin
101     */
102    public String getDescription() {
103        return description;
104    }
105
106    /**
107     * Get the plugin associated to this replacer
108     *
109     * @return the associated plugin
110     */
111    public T getPlugin() {
112        return plugin;
113    }
114
115    /**
116     * Check whether the replacer is enabled
117     *
118     * @return whether the replacer is enabled
119     */
120    public boolean isEnabled() {
121        return enabled;
122    }
123
124    /**
125     * Disable the replacer
126     */
127    public void disable() {
128        enabled = false;
129    }
130
131    /**
132     * Enables the replacer
133     */
134    public void enable() {
135        enabled = true;
136    }
137
138    @Override
139    public String toString() {
140        return ToStringBuilder.reflectionToString(this);
141    }
142}
