001package com.github.games647.scoreboardstats.variables.defaults;
002
003import com.github.games647.scoreboardstats.variables.ReplaceEvent;
004import com.github.games647.scoreboardstats.variables.VariableReplaceAdapter;
005
006import java.lang.reflect.Field;
007import java.lang.reflect.Method;
008import java.lang.reflect.Modifier;
009
010import org.bukkit.Bukkit;
011import org.bukkit.entity.Player;
012import org.bukkit.plugin.Plugin;
013
014/**
015 * Replace the ping variable.
016 */
017public class PlayerPingVariable extends VariableReplaceAdapter<Plugin> {
018
019    private Method getHandleMethod;
020    private Field pingField;
021
022    public PlayerPingVariable() {
023        super(null, "ping");
024    }
025
026    @Override
027    public void onReplace(Player player, String variable, ReplaceEvent replaceEvent) {
028        replaceEvent.setScore(getReflectionPing(player));
029    }
030
031    private int getReflectionPing(Player player) {
032        try {
033            if (getHandleMethod == null) {
034                getHandleMethod = player.getClass().getDeclaredMethod("getHandle");
035                //disable java security check. This will speed it a little
036                getHandleMethod.setAccessible(true);
037            }
038
039            final Object entityPlayer = getHandleMethod.invoke(player);
040
041            if (pingField == null) {
042                if (Bukkit.getVersion().contains("MCPC")) {
043                    //MCPC has a remapper, but it doesn't work if we get the class dynamic
044                    setMCPCPing(entityPlayer);
045                } else {
046                    pingField = entityPlayer.getClass().getDeclaredField("ping");
047                    //disable java security check. This will speed it a little
048                    pingField.setAccessible(true);
049                }
050            }
051
052            //returns the found int value
053            return pingField.getInt(entityPlayer);
054        } catch (Exception ex) {
055            //Forward the exception to replaceManager
056            throw new RuntimeException(ex);
057        }
058    }
059
060    private void setMCPCPing(Object entityPlayer) {
061        //this isn't a secure, because it detects the ping variable by the ordering
062        //a remaping (deobfuscate the variables) would work, but it won't be forwardcompatible
063        Class<?> lastType = null;
064        Field lastIntField = null;
065        for (Field field : entityPlayer.getClass().getDeclaredFields()) {
066            if (field.getType() == Integer.TYPE
067                    && Modifier.isPublic(field.getModifiers())
068                    && lastType == Boolean.TYPE) {
069                lastIntField = field;
070                continue;
071            }
072
073            if (field.getType() == Boolean.TYPE && lastIntField != null) {
074                pingField = lastIntField;
075                //disable java security check. This will speed it a little
076                pingField.setAccessible(true);
077                break;
078            }
079
080            lastIntField = null;
081            lastType = field.getType();
082        }
083    }
084}
