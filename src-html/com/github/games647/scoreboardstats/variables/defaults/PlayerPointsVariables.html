001package com.github.games647.scoreboardstats.variables.defaults;
002
003import com.github.games647.scoreboardstats.variables.ReplaceEvent;
004import com.github.games647.scoreboardstats.variables.ReplaceManager;
005import com.github.games647.scoreboardstats.variables.VariableReplaceAdapter;
006
007import org.black_ixx.playerpoints.PlayerPoints;
008import org.black_ixx.playerpoints.event.PlayerPointsEvent;
009import org.bukkit.Bukkit;
010import org.bukkit.entity.Player;
011import org.bukkit.event.EventHandler;
012import org.bukkit.event.Listener;
013
014/**
015 * Represents a replacer for the Plugin PlayerPoints
016 */
017public class PlayerPointsVariables extends VariableReplaceAdapter<PlayerPoints> implements Listener {
018
019    private final ReplaceManager replaceManager;
020
021    public PlayerPointsVariables(ReplaceManager replaceManager) {
022        super((PlayerPoints) Bukkit.getPluginManager().getPlugin("PlayerPoints"), "points");
023
024        this.replaceManager = replaceManager;
025    }
026
027    @Override
028    public void onReplace(Player player, String variable, ReplaceEvent replaceEvent) {
029        final int balance = getPlugin().getAPI().look(player.getUniqueId());
030        replaceEvent.setScore(balance);
031        replaceEvent.setConstant(true);
032    }
033
034    @EventHandler(ignoreCancelled = true)
035    public void onPointsChange(PlayerPointsEvent changeEvent) {
036        final Player player = Bukkit.getPlayer(changeEvent.getPlayerId());
037        if (player != null) {
038            final int lastBal = getPlugin().getAPI().look(changeEvent.getPlayerId());
039            replaceManager.updateScore(player, "points", lastBal + changeEvent.getChange());
040        }
041    }
042}
