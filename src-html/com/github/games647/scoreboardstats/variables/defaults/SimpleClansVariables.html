001package com.github.games647.scoreboardstats.variables.defaults;
002
003import com.github.games647.scoreboardstats.Version;
004import com.github.games647.scoreboardstats.variables.ReplaceEvent;
005import com.github.games647.scoreboardstats.variables.UnsupportedPluginException;
006import com.github.games647.scoreboardstats.variables.VariableReplaceAdapter;
007
008import net.sacredlabyrinth.phaed.simpleclans.Clan;
009import net.sacredlabyrinth.phaed.simpleclans.ClanPlayer;
010import net.sacredlabyrinth.phaed.simpleclans.SimpleClans;
011import net.sacredlabyrinth.phaed.simpleclans.managers.ClanManager;
012
013import org.bukkit.Bukkit;
014import org.bukkit.entity.Player;
015import org.bukkit.util.NumberConversions;
016
017/**
018 * Replace all variables that are associated with the SimpleClans plugin
019 *
020 * @see SimpleClans
021 * @see ClanPlayer
022 * @see Clan
023 * @see ClanManager
024 */
025public class SimpleClansVariables extends VariableReplaceAdapter<SimpleClans> {
026
027    private final ClanManager clanManager;
028
029    /**
030     * Creates a new SimpleClans replacer
031     */
032    public SimpleClansVariables() {
033        super((SimpleClans) Bukkit.getPluginManager().getPlugin("SimpleClans")
034                , "kills", "deaths", "kdr"
035                , "member", "clan_kdr", "rivals", "allies"
036                , "clan_money", "clan_kills", "allies_total", "members_online");
037
038        final String version = getPlugin().getDescription().getVersion();
039        if (Version.compare("2.5", version) >= 0) {
040            clanManager = getPlugin().getClanManager();
041        } else {
042            throw new UnsupportedPluginException("SimpleClans under version 2.5 are not supported");
043        }
044    }
045
046    @Override
047    public void onReplace(Player player, String variable, ReplaceEvent replaceEvent) {
048        //If simpleclans doesn't track the player yet return -1
049        final ClanPlayer clanPlayer = clanManager.getClanPlayer(player);
050        if (clanPlayer == null) {
051            replaceEvent.setScore(-1);
052            return;
053        }
054
055        if ("kills".equals(variable)) {
056            final int civilianKills = clanPlayer.getCivilianKills();
057            final int neutralKills = clanPlayer.getNeutralKills();
058            final int rivalKills = clanPlayer.getRivalKills();
059            //count all kill types
060            replaceEvent.setScore(civilianKills + neutralKills + rivalKills);
061        } else if ("deaths".equals(variable)) {
062            replaceEvent.setScore(clanPlayer.getDeaths());
063        } else if ("kdr".equals(variable)) {
064            replaceEvent.setScore(Math.round(clanPlayer.getKDR()));
065        } else {
066            //Check if the player has a clan
067            final Clan clan = clanPlayer.getClan();
068            if (clan == null) {
069                replaceEvent.setScore(-1);
070                return;
071            }
072
073            if ("member".equals(variable)) {
074                replaceEvent.setScore(clan.getMembers().size());
075            } else if ("clan_kdr".equals(variable)) {
076                replaceEvent.setScore(Math.round(clan.getTotalKDR()));
077            }
078            if ("clan_money".equals(variable)) {
079                replaceEvent.setScore(NumberConversions.round(clan.getBalance()));
080            } else if ("rivals".equals(variable)) {
081                replaceEvent.setScore(clan.getRivals().size());
082            } else if ("allies".equals(variable)) {
083                replaceEvent.setScore(clan.getAllies().size());
084            } else if ("members_online".equals(variable)) {
085                replaceEvent.setScore(clan.getOnlineMembers().size());
086            } else if ("allies_total".equals(variable)) {
087                replaceEvent.setScore(clan.getAllAllyMembers().size());
088            } else if ("clan_kills".equals(variable)) {
089                final int civilianKills = clan.getTotalCivilian();
090                final int neutralKills = clan.getTotalNeutral();
091                final int rivalKills = clan.getTotalNeutral();
092                //count all kill types
093                replaceEvent.setScore(civilianKills + neutralKills + rivalKills);
094            }
095        }
096    }
097}
