001package com.github.games647.scoreboardstats.variables.defaults;
002
003import com.github.games647.scoreboardstats.variables.ReplaceEvent;
004import com.github.games647.scoreboardstats.variables.ReplaceManager;
005import com.github.games647.scoreboardstats.variables.VariableReplaceAdapter;
006import com.herocraftonline.heroes.Heroes;
007import com.herocraftonline.heroes.api.events.ClassChangeEvent;
008import com.herocraftonline.heroes.api.events.HeroChangeLevelEvent;
009import com.herocraftonline.heroes.api.events.HeroRegainManaEvent;
010import com.herocraftonline.heroes.api.events.SkillUseEvent;
011import com.herocraftonline.heroes.characters.CharacterManager;
012import com.herocraftonline.heroes.characters.Hero;
013import com.herocraftonline.heroes.characters.classes.HeroClass;
014
015import org.bukkit.Bukkit;
016import org.bukkit.entity.Player;
017import org.bukkit.event.EventHandler;
018import org.bukkit.event.Listener;
019
020/**
021 * Replace all variables that are associated with the heroes plugin
022 */
023public class HeroesVariables extends VariableReplaceAdapter<Heroes> implements Listener {
024
025    private final ReplaceManager replaceManager;
026    private final CharacterManager characterManager;
027
028    /**
029     * Creates a new heroes replacer
030     *
031     * @param replaceManager the replace manager from ScoreboardStats
032     */
033    public HeroesVariables(ReplaceManager replaceManager) {
034        super((Heroes) Bukkit.getPluginManager().getPlugin("Heroes"), "mana", "level", "max_mana");
035
036        this.replaceManager = replaceManager;
037        characterManager = getPlugin().getCharacterManager();
038    }
039
040    @Override
041    public void onReplace(Player player, String variable, ReplaceEvent replaceEvent) {
042        final Hero hero = characterManager.getHero(player);
043        replaceEvent.setConstant(true);
044        if ("mana".equals(variable)) {
045            replaceEvent.setScore(hero.getMana());
046        } else if ("level".equals(variable)) {
047            replaceEvent.setScore(hero.getLevel());
048        } else if ("max_mana".equals(variable)) {
049            replaceEvent.setScore(hero.getMaxMana());
050        }
051    }
052
053    @EventHandler(ignoreCancelled = true)
054    public void onLevelChange(HeroChangeLevelEvent levelChangeEvent) {
055        final Player player = levelChangeEvent.getHero().getPlayer();
056        final HeroClass heroClass = levelChangeEvent.getHeroClass();
057        final int newLevel = levelChangeEvent.getTo();
058
059        replaceManager.updateScore(player, "level", newLevel);
060        replaceManager.updateScore(player, "max_mana", (int) heroClass.getMaxManaPerLevel() * newLevel);
061    }
062
063    @EventHandler(ignoreCancelled = true)
064    public void onManaRegain(HeroRegainManaEvent regainManaEvent) {
065        final Hero hero = regainManaEvent.getHero();
066        replaceManager.updateScore(hero.getPlayer(), "mana", hero.getMana());
067    }
068
069    @EventHandler(ignoreCancelled = true)
070    public void onSkillUse(SkillUseEvent skillUseEvent) {
071        final Hero hero = skillUseEvent.getHero();
072        replaceManager.updateScore(hero.getPlayer(), "mana", hero.getMana() - skillUseEvent.getManaCost());
073    }
074
075    @EventHandler(ignoreCancelled = true)
076    public void onClassChange(ClassChangeEvent classChangeEvent) {
077        final Hero hero = classChangeEvent.getHero();
078        final int maxMana = (int) classChangeEvent.getTo().getMaxManaPerLevel();
079        replaceManager.updateScore(hero.getPlayer(), "max_mana", maxMana);
080    }
081}
