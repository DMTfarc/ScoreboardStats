001package com.github.games647.scoreboardstats.variables;
002
003import java.util.Iterator;
004import java.util.Map;
005
006import org.bukkit.event.EventHandler;
007import org.bukkit.event.Listener;
008import org.bukkit.event.server.PluginDisableEvent;
009import org.bukkit.event.server.PluginEnableEvent;
010import org.bukkit.plugin.Plugin;
011
012/**
013 * Keeps track of plugin disables and enables. It will register default replacers
014 * back again or removes replacers of disabled plugins.
015 */
016public class PluginListener implements Listener {
017
018    private final ReplaceManager replaceManager;
019
020    public PluginListener(ReplaceManager replaceManager) {
021        this.replaceManager = replaceManager;
022    }
023
024    /**
025     * Check for disabled plugin to re add the associated replacer
026     *
027     * @param enableEvent the enable event
028     */
029    @EventHandler
030    public void onPluginEnable(PluginEnableEvent enableEvent) {
031        //Register the listener back again if the plugin is available
032        final String enablePluginName = enableEvent.getPlugin().getName();
033        final Map<Class<? extends VariableReplaceAdapter<?>>, String> defaults = replaceManager.getDefaults();
034        for (Map.Entry<Class<? extends VariableReplaceAdapter<?>>, String> entry : defaults.entrySet()) {
035            final String pluginName = entry.getValue();
036            if (enablePluginName.equals(entry.getValue())) {
037                replaceManager.registerDefault(entry.getKey(), pluginName);
038            }
039        }
040    }
041
042    /**
043     * Check for disabled plugin to remove the associated replacer
044     *
045     * @param disableEvent the disable event
046     */
047    @EventHandler
048    public void onPluginDisable(PluginDisableEvent disableEvent) {
049        //Remove the listener if the associated plugin was disabled
050        final String disablePluginName = disableEvent.getPlugin().getName();
051
052        final Map<String, VariableReplaceAdapter<? extends Plugin>> specificReplacers = replaceManager.getReplacers();
053        final Iterator<VariableReplaceAdapter<? extends Plugin>> iterator = specificReplacers.values().iterator();
054        while (iterator.hasNext()) {
055            final Plugin plugin = iterator.next().getPlugin();
056            if (plugin != null && plugin.getName().equals(disablePluginName)) {
057                iterator.remove();
058            }
059        }
060    }
061}
