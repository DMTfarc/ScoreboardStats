001package com.github.games647.scoreboardstats.variables;
002
003import com.google.common.collect.Lists;
004
005import java.util.List;
006
007import org.bukkit.entity.Player;
008import org.bukkit.plugin.Plugin;
009
010/**
011 * Represents a wrapper in order to support the usage of the old Replaceable
012 */
013public class LegacyReplaceWrapper extends VariableReplaceAdapter<Plugin> {
014
015    private final Replaceable oldReplacer;
016    private List<String> variables = Lists.newArrayList();
017
018    public LegacyReplaceWrapper(Plugin plugin, @SuppressWarnings("deprecation") Replaceable oldReplacer) {
019        super(plugin);
020
021        this.oldReplacer = oldReplacer;
022    }
023
024    @Override
025    public List<String> getVariables() {
026        return variables;
027    }
028
029    @Override
030    @SuppressWarnings("deprecation")
031    public void onReplace(Player player, String variable, ReplaceEvent replaceEvent) {
032        final int scoreValue = oldReplacer.getScoreValue(player, '%' + variable + '%');
033        if (scoreValue != Replaceable.UNKOWN_VARIABLE) {
034            replaceEvent.setScore(scoreValue);
035        }
036    }
037
038    @Override
039    public int hashCode() {
040        //make it possible to remove them using the Replaceable instance
041        return oldReplacer.hashCode();
042    }
043
044    @Override
045    public boolean equals(Object other) {
046        //make it possible to remove them using the Replaceable instance
047        return oldReplacer.equals(other);
048    }
049
050    @Override
051    public String toString() {
052        return oldReplacer.toString();
053    }
054}
