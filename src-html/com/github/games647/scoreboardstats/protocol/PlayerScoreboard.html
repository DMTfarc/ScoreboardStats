001package com.github.games647.scoreboardstats.protocol;
002
003import com.google.common.base.Preconditions;
004import com.google.common.collect.ImmutableSet;
005import com.google.common.collect.Maps;
006
007import java.util.Collection;
008import java.util.Map;
009
010import org.bukkit.entity.Player;
011
012/**
013 * Represents the scoreboard overview in a server-side perspective.
014 */
015public class PlayerScoreboard {
016
017    private final Player player;
018
019    private final Map<String, Objective> objectivesByName = Maps.newHashMap();
020
021    private Objective sidebarObjective;
022
023    /**
024     * Creates a new scoreboard for specific player.
025     *
026     * @param player the player for the scoreboard
027     */
028    public PlayerScoreboard(Player player) {
029        this.player = player;
030    }
031
032    /**
033     * Creates a new sidebar objective.
034     *
035     * @param objectiveName the objective name. have to be unique
036     * @param displayName the displayed name
037     * @param replace overwrite the sidebar slot. Conflicts if you overwrite scoreboards from other plugins
038     * @return the created objective
039     * @throws NullPointerException name is null
040     * @throws NullPointerException displayName is null
041     * @throws IllegalArgumentException name is longer than 16 characters
042     * @throws IllegalArgumentException displayname is longer than 32 characters
043     * @throws IllegalStateException if there is already a objective with that name
044     * @throws IllegalStateException if there is already a sidebar objective active
045     */
046    public Objective createSidebarObjective(String objectiveName, String displayName, boolean replace)
047            throws NullPointerException, IllegalArgumentException, IllegalStateException {
048        Preconditions.checkNotNull(objectiveName, "objective name cannot be null");
049        Preconditions.checkNotNull(displayName, "display name cannot be null");
050
051        Preconditions.checkArgument(objectiveName.length() <= 16, "objective name is longer than 16 characters");
052        Preconditions.checkArgument(displayName.length() <= 32, "display name is longer than 32 characters");
053
054        if (!replace) {
055            Preconditions.checkState(sidebarObjective == null, "There is already an sidebar objective");
056        }
057
058        if (objectivesByName.containsKey(objectiveName)) {
059            //the objecive already exits. I assume that no other use this unique name
060            //so we expect that a other sidebar was showing
061            final Objective objective = objectivesByName.get(objectiveName);
062            PacketFactory.sendDisplayPacket(objective);
063            sidebarObjective = objective;
064            return objective;
065        }
066
067        final Objective objective = new Objective(this, objectiveName, displayName);
068        sidebarObjective = objective;
069        objectivesByName.put(objectiveName, objective);
070        return objective;
071    }
072
073    /**
074     * Gets the current sidebar objective
075     *
076     * @return current sidebar objective or null
077     */
078    public Objective getSidebarObjective() {
079        return sidebarObjective;
080    }
081
082    /**
083     * Gets all objectives that the player have as immutable collection.
084     *
085     * @return the objectives
086     */
087    public Collection<Objective> getObjectives() {
088        return ImmutableSet.copyOf(objectivesByName.values());
089    }
090
091    /**
092     * Gets the owner of this scoreboard
093     *
094     * @return the owner of the scoreboard
095     */
096    public Player getPlayer() {
097        return player;
098    }
099
100    protected void addObjective(String objectiveName, String displayName) {
101        objectivesByName.put(objectiveName, new Objective(this, objectiveName, displayName, false));
102    }
103
104    protected Objective getObjective(String name) {
105        return objectivesByName.get(name);
106    }
107
108    protected void removeObjective(String objectiveName) {
109        objectivesByName.remove(objectiveName);
110        if (sidebarObjective != null && sidebarObjective.getName().equals(objectiveName)) {
111            clearSidebarObjective();
112        }
113    }
114
115    protected void setSidebarObjective(String objectiveName) {
116        if (objectiveName.isEmpty()) {
117            clearSidebarObjective();
118        } else {
119            sidebarObjective = objectivesByName.get(objectiveName);
120        }
121    }
122
123    protected void clearSidebarObjective() {
124        sidebarObjective = null;
125    }
126
127    protected void resetScore(String scoreName) {
128        /*
129         * Very weird that minecraft always ignore the name of the parent objective and
130         * will remove the score from the complete scoreboard
131         */
132        for (Objective entry : objectivesByName.values()) {
133            entry.items.remove(scoreName);
134        }
135    }
136
137    protected void createOrUpdateScore(String scoreName, String parent, int score) {
138        final Objective objective = objectivesByName.get(parent);
139        if (objective != null) {
140            final Item item = new Item(objective, scoreName, score, false);
141            //This automatically replace the old one
142            objective.items.put(scoreName, item);
143        }
144    }
145}
