001package com.github.games647.scoreboardstats.protocol;
002
003import com.comphenix.protocol.PacketType;
004import com.comphenix.protocol.events.PacketAdapter;
005import com.comphenix.protocol.events.PacketContainer;
006import com.comphenix.protocol.events.PacketEvent;
007import com.comphenix.protocol.reflect.StructureModifier;
008
009import org.bukkit.entity.Player;
010import org.bukkit.plugin.Plugin;
011import org.bukkit.scoreboard.DisplaySlot;
012
013/**
014 * Listening all outgoing packets and check + handle for possibly client crash cases.
015 * This Listener should only read and listen to relevant packets.
016 *
017 * Protocol specifications can be found here http://wiki.vg/Protocol
018*
019 * @see PacketFactory
020 * @see PacketAdapter
021 */
022public class PacketListener extends PacketAdapter {
023
024    //Shorter access
025    private static final PacketType DISPLAY_TYPE = PacketType.Play.Server.SCOREBOARD_DISPLAY_OBJECTIVE;
026    private static final PacketType OBJECTIVE_TYPE = PacketType.Play.Server.SCOREBOARD_OBJECTIVE;
027    private static final PacketType SCORE_TYPE = PacketType.Play.Server.SCOREBOARD_SCORE;
028
029    private final PacketSbManager manager;
030
031    /**
032     * Creates a new packet listener
033     *
034     * @param plugin plugin for registration into ProtcolLib
035     * @param manager packet manager instance
036     */
037    public PacketListener(Plugin plugin, PacketSbManager manager) {
038        super(plugin, DISPLAY_TYPE, OBJECTIVE_TYPE, SCORE_TYPE);
039
040        this.manager = manager;
041    }
042
043    @Override
044    public void onPacketSending(PacketEvent event) {
045        if (event.isCancelled()) {
046            return;
047        }
048
049        final PacketType packetType = event.getPacketType();
050        if (packetType.equals(SCORE_TYPE)) {
051            handleScorePacket(event);
052        } else if (packetType.equals(OBJECTIVE_TYPE)) {
053            handleObjectivePacket(event);
054        } else if (packetType.equals(DISPLAY_TYPE)) {
055            handleDisplayPacket(event);
056        }
057    }
058
059    private void handleScorePacket(PacketEvent event) {
060        final Player player = event.getPlayer();
061        final PacketContainer packet = event.getPacket();
062
063        final String scoreName = packet.getStrings().read(0);
064        final String parent = packet.getStrings().read(1);
065        final int score = packet.getIntegers().read(0);
066
067        //state id
068        final StructureModifier<Enum> enumModifier = packet.getSpecificModifier(Enum.class);
069        final Enum<?> scoreboardActions = enumModifier.readSafely(0);
070
071        State action;
072        if (scoreboardActions == null) {
073            //old system
074            action = State.fromId(packet.getIntegers().read(1));
075        } else {
076            action = State.fromId(scoreboardActions.ordinal());
077        }
078
079        //Packet receiving validation
080        if (action == State.CREATE && parent.length() > 16) {
081            //Invalid packet
082            return;
083        }
084
085        final PlayerScoreboard scoreboard = manager.getScoreboard(player);
086        //scores actually only have two state id, because these
087        if (action == State.CREATE) {
088            scoreboard.createOrUpdateScore(scoreName, parent, score);
089        } else if (action == State.REMOVE) {
090            scoreboard.resetScore(scoreName);
091        }
092    }
093
094    private void handleObjectivePacket(PacketEvent event) {
095        final Player player = event.getPlayer();
096        final PacketContainer packet = event.getPacket();
097
098        final String objectiveName = packet.getStrings().read(0);
099        //Can be empty
100        final String displayName = packet.getStrings().read(1);
101        final State action = State.fromId(packet.getIntegers().read(0));
102
103        //Packet receiving validation
104        if (objectiveName.length() > 16 || displayName.length() > 32) {
105            //Invalid packet
106            return;
107        }
108
109        final PlayerScoreboard scoreboard = manager.getScoreboard(player);
110        final Objective objective = scoreboard.getObjective(objectiveName);
111        if (action == State.CREATE) {
112            scoreboard.addObjective(objectiveName, displayName);
113        } else if (objective != null) {
114            //Could cause a NPE at the client if the objective wasn't found
115            if (action == State.REMOVE) {
116                scoreboard.removeObjective(objectiveName);
117            } else if (action == State.UPDATE_TITLE) {
118                objective.setDisplayName(displayName, false);
119            }
120        }
121    }
122
123    private void handleDisplayPacket(PacketEvent event) {
124        final Player player = event.getPlayer();
125        final PacketContainer packet = event.getPacket();
126
127        //Can be empty; if so it would just clear the slot
128        final String objectiveName = packet.getStrings().read(0);
129        final DisplaySlot slot = SlotTransformer.fromId(packet.getIntegers().read(0));
130
131        //Packet receiving validation
132        if (slot == null || objectiveName.length() > 16) {
133            return;
134        }
135
136        final PlayerScoreboard scoreboard = manager.getScoreboard(player);
137        if (slot == DisplaySlot.SIDEBAR) {
138            scoreboard.setSidebarObjective(objectiveName);
139        } else {
140            final Objective sidebarObjective = scoreboard.getSidebarObjective();
141            if (sidebarObjective != null && sidebarObjective.getName().equals(objectiveName)) {
142                scoreboard.clearSidebarObjective();
143            }
144        }
145    }
146}
