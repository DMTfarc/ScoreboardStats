001package com.github.games647.scoreboardstats.protocol;
002
003import com.comphenix.protocol.ProtocolLibrary;
004import com.github.games647.scoreboardstats.Lang;
005import com.github.games647.scoreboardstats.SbManager;
006import com.github.games647.scoreboardstats.ScoreboardStats;
007import com.github.games647.scoreboardstats.config.Settings;
008import com.github.games647.scoreboardstats.variables.ReplaceEvent;
009import com.github.games647.scoreboardstats.variables.UnknownVariableException;
010
011import java.util.Iterator;
012import java.util.Map;
013import java.util.WeakHashMap;
014
015import org.bukkit.entity.Player;
016
017/**
018 * Manage the scoreboards with packet-use
019 */
020public class PacketSbManager extends SbManager {
021
022    private final Map<Player, PlayerScoreboard> scoreboards = new WeakHashMap<Player, PlayerScoreboard>(50);
023
024    /**
025     * Creates a new scoreboard manager for the packet system.
026     *
027     * @param plugin ScoreboardStats instance
028     */
029    public PacketSbManager(ScoreboardStats plugin) {
030        super(plugin);
031
032        ProtocolLibrary.getProtocolManager().addPacketListener(new PacketListener(plugin, this));
033    }
034
035    /**
036     * Gets the scoreboard from a player.
037     *
038     * @param player who owns the scoreboard
039     * @return the scoreboard instance
040     */
041    public PlayerScoreboard getScoreboard(Player player) {
042        PlayerScoreboard scoreboard = scoreboards.get(player);
043        if (scoreboard == null) {
044            //lazy loading due potenially performance issues
045            scoreboard = new PlayerScoreboard(player);
046            scoreboards.put(player, scoreboard);
047        }
048
049        return scoreboard;
050    }
051
052    @Override
053    public void sendUpdate(Player player) {
054        final Objective sidebar = getScoreboard(player).getSidebarObjective();
055        if (sidebar == null) {
056            createScoreboard(player);
057        } else {
058            sendUpdate(player, false);
059        }
060    }
061
062    @Override
063    public void unregisterAll() {
064        super.unregisterAll();
065
066        scoreboards.clear();
067    }
068
069    @Override
070    public void unregister(Player player) {
071        final PlayerScoreboard scoreboard = scoreboards.get(player);
072        for (Objective objective : scoreboard.getObjectives()) {
073            final String objectiveName = objective.getName();
074            if (objectiveName.startsWith(SB_NAME)) {
075                objective.unregister();
076            }
077        }
078    }
079
080    @Override
081    public void createScoreboard(Player player) {
082        final PlayerScoreboard scoreboard = getScoreboard(player);
083        final Objective oldObjective = scoreboard.getSidebarObjective();
084        if (!isValid(player) || oldObjective != null && !TEMP_SB_NAME.equals(oldObjective.getName())) {
085            //Check if another scoreboard is showing
086            return;
087        }
088
089        scoreboard.createSidebarObjective(SB_NAME, Settings.getTitle(), true);
090        sendUpdate(player, true);
091
092        //Schedule the next tempscoreboard show
093        scheduleShowTask(player, true);
094    }
095
096    @Override
097    public void createTopListScoreboard(Player player) {
098        final PlayerScoreboard scoreboard = getScoreboard(player);
099        final Objective oldObjective = scoreboard.getSidebarObjective();
100        if (!isValid(player) || oldObjective == null
101                || !oldObjective.getName().startsWith(SB_NAME)) {
102            //Check if another scoreboard is showing
103            return;
104        }
105
106        Objective objective = scoreboard.getObjective(TEMP_SB_NAME);
107        if (objective != null) {
108            //It's better to send an unregister and let the client handle the remove than sending up to 15
109            //item remove packets
110            objective.unregister();
111        }
112
113        //We are checking if another object is shown. If it's our scoreboard the code will continue to this
114        //were the force the replacement, because the scoreboard management in minecraft right now is sync,
115        //so we don't expect any crashes by other plugins.
116        objective = scoreboard.createSidebarObjective(TEMP_SB_NAME, Settings.getTempTitle(), true);
117
118        //Colorize and send all elements
119        for (Map.Entry<String, Integer> entry : plugin.getStatsDatabase().getTop()) {
120            final String scoreName = Settings.getTempColor() + entry.getKey();
121            sendScore(objective, scoreName, entry.getValue());
122        }
123
124        //schedule the next normal scoreboard show
125        scheduleShowTask(player, false);
126    }
127
128    @Override
129    public void update(Player player, String itemName, int newScore) {
130        final PlayerScoreboard scoreboard = scoreboards.get(player);
131        if (scoreboard != null) {
132            final Objective objective = scoreboard.getObjective(SB_NAME);
133            if (objective != null) {
134                sendScore(objective, itemName, newScore);
135            }
136        }
137    }
138
139    @Override
140    protected void sendUpdate(Player player, boolean complete) {
141        final Objective sidebar = getScoreboard(player).getSidebarObjective();
142        if (SB_NAME.equals(sidebar.getName())) {
143            final Iterator<Map.Entry<String, String>> iter = Settings.getItems();
144            while (iter.hasNext()) {
145                final Map.Entry<String, String> entry = iter.next();
146                final String title = entry.getKey();
147                final String variable = entry.getValue();
148
149                try {
150                    final ReplaceEvent replaceEvent = replaceManager.getScore(player, variable, title, 0, complete);
151                    if (replaceEvent.isModified()) {
152                        sendScore(sidebar, title, replaceEvent.getScore());
153                    }
154                } catch (UnknownVariableException ex) {
155                    //Remove the variable becaue we can't replace it
156                    iter.remove();
157
158                    plugin.getLogger().info(Lang.get("unknownVariable", variable));
159                }
160            }
161        }
162    }
163
164    private void sendScore(Objective objective, String title, int value) {
165        final Item item = objective.getItem(title);
166        if (item == null) {
167            objective.registerItem(title, value);
168        } else {
169            item.setScore(value);
170        }
171    }
172}
