001package com.github.games647.scoreboardstats.protocol;
002
003import com.comphenix.protocol.PacketType;
004import com.comphenix.protocol.ProtocolLibrary;
005import com.comphenix.protocol.ProtocolManager;
006import com.comphenix.protocol.events.PacketContainer;
007import com.comphenix.protocol.reflect.StructureModifier;
008
009import java.lang.reflect.InvocationTargetException;
010import java.util.logging.Level;
011import java.util.logging.Logger;
012
013/**
014 * Creates the specific packets and send it with help of ProtocolLib.
015 *
016 * Protocol specifications can be found here http://wiki.vg/Protocol
017 *
018 * @see PacketListener
019 * @see ProtocolManager
020 */
021public class PacketFactory {
022
023    private static final int SIDEBAR_SLOT = 1;
024
025    private static final ProtocolManager PROTOCOL_MANAGER = ProtocolLibrary.getProtocolManager();
026
027    /**
028     * Sends a new scoreboard item packet.
029     *
030     * @param item the scoreboard item
031     * @param state whether the item should be send as removed or created/updated
032     */
033    public static void sendPacket(Item item, State state) {
034        final PacketContainer scorePacket = PROTOCOL_MANAGER
035                .createPacket(PacketType.Play.Server.SCOREBOARD_SCORE, true);
036        //max length 16 and since 1.7 UTF-8 instead of UTF-16
037        scorePacket.getStrings().write(0, item.getScoreName());
038        scorePacket.getStrings().write(1, item.getParent().getName());
039
040        if (State.REMOVE != state) {
041            //Only need these if the score will be updated or created
042            scorePacket.getIntegers().write(0, item.getScore());
043        }
044
045        //state id
046        final StructureModifier<Enum> enumModifier = scorePacket.getSpecificModifier(Enum.class);
047        final Enum<?> scoreboardActions = enumModifier.readSafely(0);
048        if (scoreboardActions == null) {
049            //old system
050            scorePacket.getIntegers().write(1, state.ordinal());
051        } else {
052            final Enum action = scoreboardActions.getClass().getEnumConstants()[state.ordinal()];
053            enumModifier.write(0, action);
054        }
055
056        try {
057            //false so we don't listen to our own packets
058            PROTOCOL_MANAGER.sendServerPacket(item.getOwner(), scorePacket, false);
059        } catch (InvocationTargetException ex) {
060            //just log it for now.
061            Logger.getLogger("ScoreboardStats").log(Level.SEVERE, null, ex);
062        }
063    }
064
065    /**
066     * Sends a new scoreboard objective packet and a display packet (if the
067     * objective was created)
068     *
069     * @param objective the scoreboard objective
070     * @param state whether the objective was created, updated (displayname) or removed
071     */
072    public static void sendPacket(Objective objective, State state) {
073        final PacketContainer objectivePacket = PROTOCOL_MANAGER
074                .createPacket(PacketType.Play.Server.SCOREBOARD_OBJECTIVE, true);
075//        max length 16 and since 1.7 UTF-8 instead of UTF-16
076        objectivePacket.getStrings().write(0, objective.getName());
077
078        if (state != State.REMOVE) {
079            //only send the title if needed, so while creating the objective or update the title
080            //max length 32 and since 1.7 UTF-8 instead of UTF-16
081            objectivePacket.getStrings().write(1, objective.getDisplayName());
082            //introduced in 1.8 don't fail on versions for 1.7 or below
083            objectivePacket.getStrings().writeSafely(2, "integer");
084        }
085
086        //state id
087        objectivePacket.getIntegers().write(0, state.ordinal());
088
089        try {
090            //false so we don't listen to our own packets
091            PROTOCOL_MANAGER.sendServerPacket(objective.getOwner(), objectivePacket, false);
092            sendDisplayPacket(objective);
093        } catch (InvocationTargetException ex) {
094            //just log it for now.
095            Logger.getLogger("ScoreboardStats").log(Level.SEVERE, null, ex);
096        }
097    }
098
099    /**
100     * Set the sidebar slot to this objective
101     *
102     * @param objective the displayed objective, if getName() is empty it will just clear the sidebar
103     */
104    public static void sendDisplayPacket(Objective objective) {
105        final PacketContainer displayPacket = PROTOCOL_MANAGER
106                .createPacket(PacketType.Play.Server.SCOREBOARD_DISPLAY_OBJECTIVE, true);
107        //Can be empty to clear the sidebar slot
108        //max length 16 and since 1.7 UTF-8 instead of UTF-16
109        displayPacket.getStrings().write(0, objective.getName());
110
111        displayPacket.getIntegers().write(0, SIDEBAR_SLOT);
112
113        try {
114            //false so we don't listen to our own packets
115            PROTOCOL_MANAGER.sendServerPacket(objective.getOwner(), displayPacket, false);
116        } catch (InvocationTargetException ex) {
117            //just log it for now.
118            Logger.getLogger("ScoreboardStats").log(Level.SEVERE, null, ex);
119        }
120    }
121
122    private PacketFactory() {
123        //Utility class
124    }
125}
