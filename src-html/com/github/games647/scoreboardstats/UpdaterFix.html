001package com.github.games647.scoreboardstats;
002
003import java.io.File;
004
005import org.bukkit.plugin.Plugin;
006
007/**
008 * This file exists to make sure auto updater doesn't check for new updates if
009 * the user has a development version.
010 *
011 * @see Updater
012 */
013public class UpdaterFix extends Updater {
014
015    private static final int CURSE_PROJECTID = 55148;
016
017    /**
018     * Initialize the updater.
019     *
020     * @param plugin    The plugin that is checking for an update.
021     * @param file      The file that the plugin is running from, get this from getFile() from within your main class.
022     * @param download  Specify the type of update this will be.
023     */
024    public UpdaterFix(Plugin plugin, File file, boolean download) {
025        this(plugin, file, download, null);
026    }
027
028    /**
029     * Initialize the updater with the provided callback.
030     *
031     * @param plugin    The plugin that is checking for an update.
032     * @param file      The file that the plugin is running from, get this from getFile() from within your main class.
033     * @param download  Specify the type of update this will be.
034     * @param callback  The callback instance to notify when the Updater has finished
035     */
036    public UpdaterFix(Plugin plugin, File file, boolean download, UpdateCallback callback) {
037        super(plugin, CURSE_PROJECTID, file, download, callback);
038    }
039
040    @Override
041    public boolean shouldUpdate(String localVersion, String remoteVersion) {
042        //return true if remoteVersion is higher
043        return Version.compare(localVersion, remoteVersion) > 0;
044    }
045}
