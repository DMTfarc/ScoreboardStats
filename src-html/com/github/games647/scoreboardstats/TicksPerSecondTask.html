001package com.github.games647.scoreboardstats;
002
003/**
004 * Tracks how many ticks are passed per second.
005 */
006public class TicksPerSecondTask implements Runnable {
007
008    private static double lastTicks = 20.0D;
009
010    /**
011     * Get the ticks count of the last check. 20 Ticks should pass per second
012     *
013     * @return the ticks count of the last check
014     */
015    public static double getLastTicks() {
016        return lastTicks;
017    }
018
019    //the last time we updated the ticks
020    private long lastCheck;
021
022    @Override
023    public void run() {
024        final long currentTime = System.currentTimeMillis();
025        final long timeSpent = currentTime - lastCheck;
026        //update the last check
027        lastCheck = currentTime;
028
029        //how many ticks passed since the last check * 1000 to convert to seconds
030        final double tps = 20 * 1000.0D / timeSpent;
031        if (tps >= 0.0D && tps < 25.0D) {
032            //Prevent all invalid values
033            lastTicks = tps;
034        }
035    }
036}
