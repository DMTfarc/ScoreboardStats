001package com.github.games647.scoreboardstats.pvpstats;
002
003import com.github.games647.scoreboardstats.ScoreboardStats;
004import com.github.games647.scoreboardstats.config.Settings;
005
006import org.bukkit.entity.EntityType;
007import org.bukkit.entity.LivingEntity;
008import org.bukkit.entity.Player;
009import org.bukkit.event.EventHandler;
010import org.bukkit.event.Listener;
011import org.bukkit.event.entity.EntityDeathEvent;
012import org.bukkit.event.entity.PlayerDeathEvent;
013import org.bukkit.event.player.PlayerJoinEvent;
014import org.bukkit.event.player.PlayerQuitEvent;
015
016/**
017 * If enabled this class counts the kills.
018 */
019public class StatsListener implements Listener {
020
021    private final ScoreboardStats plugin;
022    private final Database database;
023
024    public StatsListener(ScoreboardStats plugin, Database database) {
025        this.plugin = plugin;
026        this.database = database;
027    }
028
029    /**
030     * Add the player account from the database in the cache.
031     *
032     * @param joinEvent the join event
033     * @see Database
034     */
035    @EventHandler
036    public void onJoin(PlayerJoinEvent joinEvent) {
037        final Player player = joinEvent.getPlayer();
038        //removing old metadata which wasn't removed (which can lead to memory leaks)
039        player.removeMetadata("player_stats", plugin);
040
041        //load the pvpstats if activated
042        database.loadAccountAsync(player);
043    }
044
045    /**
046     * Saves the stats to database if the player leaves
047     *
048     * @param quitEvent leave event
049     * @see Database
050     */
051    @EventHandler
052    public void onQuit(PlayerQuitEvent quitEvent) {
053        final Player player = quitEvent.getPlayer();
054
055        database.saveAsync(database.getCachedStats(player));
056
057        //just remove our metadata to prevent memory leaks
058        player.removeMetadata("player_stats", plugin);
059    }
060
061    /**
062     * Tracks the mob kills.
063     *
064     * @param event the death event
065     */
066    @EventHandler
067    public void onMobDeath(EntityDeathEvent event) {
068        final LivingEntity entity = event.getEntity();
069        //killer is null if it's not a player
070        final Player killer = entity.getKiller();
071
072        //Check if it's not player because we are already handling it
073        if (entity.getType() != EntityType.PLAYER && Settings.isPvpStats()
074                && Settings.isActiveWorld(entity.getWorld().getName())) {
075            final PlayerStats killercache = database.getCachedStats(killer);
076            if (killercache != null) {
077                //If the cache entry is loaded and the player isn't null, increase the mob kills
078                killercache.incrementMobKills();
079                plugin.getReplaceManager().updateScore(killer, "mob", killercache.getMobkills());
080            }
081        }
082    }
083
084    /**
085     * Tracks player deaths and kills
086     *
087     * @param deathEvent the death event.
088     */
089    @EventHandler
090    public void onDeath(PlayerDeathEvent deathEvent) {
091        final Player killed = deathEvent.getEntity();
092        //killer is null if it's not a player
093        final Player killer = killed.getKiller();
094        if (Settings.isPvpStats() && Settings.isActiveWorld(killed.getWorld().getName())) {
095            final PlayerStats killedcache = database.getCachedStats(killed);
096            if (killedcache != null) {
097                killedcache.incrementDeaths();
098                plugin.getReplaceManager().updateScore(killed, "deaths", killedcache.getDeaths());
099                plugin.getReplaceManager().updateScore(killed, "kdr", killedcache.getKdr());
100                //will reset
101                plugin.getReplaceManager().updateScore(killed, "current_streak", killedcache.getLaststreak());
102            }
103
104            final PlayerStats killercache = database.getCachedStats(killer);
105            if (killercache != null) {
106                killercache.incrementKills();
107                plugin.getReplaceManager().updateScore(killed, "deaths", killercache.getKills());
108                plugin.getReplaceManager().updateScore(killed, "kdr", killercache.getKdr());
109                //maybe the player reaches a new high score
110                plugin.getReplaceManager().updateScore(killed, "killstreak", killercache.getKillstreak());
111                plugin.getReplaceManager().updateScore(killed, "current_streak", killercache.getLaststreak());
112            }
113        }
114    }
115}
