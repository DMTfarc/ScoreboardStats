001package com.github.games647.scoreboardstats.pvpstats;
002
003import com.avaje.ebeaninternal.server.util.ClassPathReader;
004
005import org.apache.commons.lang.ArrayUtils;
006
007/**
008 * This is workaround to a bug in Java 6. eBean 2.7.3 has a bug catching up
009 * the wrong exception while scanning through the ebean.properties. Spigot
010 * fixed this with updating eBean to 2.8.1 and newer versions of eBean has
011 * fixed it completely.
012 *
013 * The actually problem is that non-latin characters cannot be read in Java 6.
014 * Java 7 has a fix for it by passing a charset, but this still cannot be used in
015 * Java 6 and eBean doesn't use it either.
016 */
017public class PathReader implements ClassPathReader {
018
019    @Override
020    public Object[] readPath(ClassLoader classLoader) {
021        //return an empty array without creating one
022        return ArrayUtils.EMPTY_OBJECT_ARRAY;
023    }
024}
