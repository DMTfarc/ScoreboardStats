001package com.github.games647.scoreboardstats;
002
003import com.github.games647.scoreboardstats.config.Settings;
004import com.google.common.collect.Maps;
005
006import java.util.Iterator;
007import java.util.Map;
008import java.util.Set;
009
010import org.apache.commons.lang.mutable.MutableInt;
011import org.bukkit.entity.Player;
012
013/**
014 * Handling all updates for a player in a performance optimized variant. This
015 * class split the updates over the ticks much smoother.
016 *
017 * @see SbManager
018 * @see com.github.games647.scoreboardstats.variables.ReplaceManager
019 */
020public class RefreshTask implements Runnable {
021
022    private final ScoreboardStats plugin;
023
024    //Prevent duplicate entries and is faster than the delay queue
025    private final Map<Player, MutableInt> queue = Maps.newHashMapWithExpectedSize(100);
026
027    private int nextGlobalUpdate = 20 * Settings.getIntervall();
028
029    /**
030     * Initialize refresh task
031     *
032     * @param instance ScoreboardStats instance
033     */
034    public RefreshTask(ScoreboardStats instance) {
035        this.plugin = instance;
036    }
037
038    @Override
039    public void run() {
040        //let the players update smoother
041        final Set<Map.Entry<Player, MutableInt>> entrySet = queue.entrySet();
042        int remainingUpdates = getNextUpdates();
043        for (final Iterator<Map.Entry<Player, MutableInt>> it = entrySet.iterator(); it.hasNext();) {
044            final Map.Entry<Player, MutableInt> entry = it.next();
045
046            final Player player = entry.getKey();
047            final MutableInt remanigTicks = entry.getValue();
048            if (remanigTicks.intValue() == 0) {
049                //We will check if the player is online and remove it from queue if not so we can prevent memory leaks
050                if (player == null || !player.isOnline()) {
051                    it.remove();
052                } else if (remainingUpdates != 0) {
053                    //Smoother refreshing; limit the updates
054                    plugin.getScoreboardManager().sendUpdate(player);
055                    remanigTicks.setValue(20 * Settings.getIntervall());
056                    remainingUpdates--;
057                }
058            } else {
059                remanigTicks.decrement();
060            }
061        }
062
063        nextGlobalUpdate--;
064        if (nextGlobalUpdate == 0) {
065            nextGlobalUpdate = 20 * Settings.getIntervall();
066            //update globals
067            plugin.getReplaceManager().updateGlobals();
068        }
069    }
070
071    /**
072     * Add a player to the queue for updating him.
073     *
074     * @param request the player that should be added.
075     * @return true if it was successfully queued
076     */
077    public boolean addToQueue(Player request) {
078        final boolean alreadyQueued = queue.containsKey(request);
079        if (!alreadyQueued) {
080            //check if it isn't already in the queue
081            queue.put(request, new MutableInt(20 * Settings.getIntervall()));
082        }
083
084        return !alreadyQueued;
085    }
086
087    /**
088     * Checks if the player is in the refresh queue.
089     *
090     * @param request player instance
091     * @return true if the player is in the refresh queue.
092     */
093    public boolean contains(Player request) {
094        return queue.containsKey(request);
095    }
096
097    /**
098     * Explicity removes the player from the refresh queue.
099     *
100     * @param request player who should be removed
101     * @return if the last entry exists
102     */
103    public boolean remove(Player request) {
104        return queue.remove(request) != null;
105    }
106
107    /**
108     * Clears the complete queue.
109     */
110    public void clear() {
111        queue.clear();
112    }
113
114    private int getNextUpdates() {
115        final int nextUpdates = queue.size() / 20;
116        if (nextUpdates <= 0) {
117            //just update minimum one player per tick. Otherwise servers with not much players
118            //won't receive any updates
119            return 1;
120        }
121
122        return nextUpdates;
123    }
124}
