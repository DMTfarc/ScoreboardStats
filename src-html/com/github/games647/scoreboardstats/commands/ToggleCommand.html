001package com.github.games647.scoreboardstats.commands;
002
003import com.github.games647.scoreboardstats.Lang;
004import com.github.games647.scoreboardstats.RefreshTask;
005import com.github.games647.scoreboardstats.ScoreboardStats;
006
007import org.bukkit.command.CommandSender;
008import org.bukkit.entity.Player;
009import org.bukkit.scoreboard.DisplaySlot;
010
011/**
012 * Change the visibility of the scoreboard
013 */
014public class ToggleCommand extends CommandHandler {
015
016    public ToggleCommand(ScoreboardStats plugin) {
017        super("toggle", plugin, "hide", "show");
018    }
019
020    @Override
021    public void onCommand(CommandSender sender, String subCommand, String... args) {
022        if (!(sender instanceof Player)) {
023            //the console cannot have a scoreboard
024            sender.sendMessage(Lang.get("noConsole"));
025            return;
026        }
027
028         //We checked that it can only be players
029        final Player player = (Player) sender;
030        final RefreshTask refreshTask = plugin.getRefreshTask();
031        if (refreshTask.contains(player)) {
032            if ("hide".equals(subCommand) || "toggle".equals(subCommand)) {
033                refreshTask.remove(player);
034                player.getScoreboard().clearSlot(DisplaySlot.SIDEBAR);
035                player.sendMessage(Lang.get("onToggle"));
036            }
037        } else if ("show".equals(subCommand) || "toggle".equals(subCommand)) {
038            player.sendMessage(Lang.get("onToggle"));
039            refreshTask.addToQueue(player);
040        }
041    }
042}
