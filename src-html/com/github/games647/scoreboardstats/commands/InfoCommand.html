001package com.github.games647.scoreboardstats.commands;
002
003import com.github.games647.scoreboardstats.ScoreboardStats;
004
005import org.bukkit.ChatColor;
006import org.bukkit.command.CommandSender;
007
008/**
009 * Show plugin informations like version number
010 */
011public class InfoCommand extends CommandHandler {
012
013    private static final String PREFIX = ChatColor.WHITE + "["
014            + ChatColor.GOLD + "%name%"
015            + ChatColor.WHITE + ']' + ChatColor.RESET;
016
017    public InfoCommand(ScoreboardStats plugin) {
018        super("info", plugin);
019    }
020
021    @Override
022    public void onCommand(CommandSender sender, String subCommand, String... args) {
023        final String pluginVersion = plugin.getDescription().getVersion();
024
025        final ChatColor color = ChatColor.DARK_AQUA;
026        final ChatColor highlightColor = ChatColor.AQUA;
027        final String message = PREFIX + color + " Running %name% v" + highlightColor + "%version%"
028                + highlightColor + ". Use /help" + color + " for help.";
029        sender.sendMessage(message.replace("%name%", plugin.getName()).replace("%version%", pluginVersion));
030    }
031}
