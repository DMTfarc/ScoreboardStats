001package com.github.games647.scoreboardstats.commands;
002
003import com.github.games647.scoreboardstats.Lang;
004import com.github.games647.scoreboardstats.ScoreboardStats;
005
006import java.util.Arrays;
007import java.util.List;
008
009import org.bukkit.ChatColor;
010import org.bukkit.command.CommandSender;
011
012/**
013 * Represents a handler for sub commands
014 */
015public abstract class CommandHandler {
016
017    protected final ScoreboardStats plugin;
018
019    private final String permission;
020    private final String description;
021    private final String subCommand;
022
023    private final List<String> aliases;
024
025    public CommandHandler(String command, ScoreboardStats plugin, String... aliases) {
026        this(command, "&cNo description", plugin, aliases);
027    }
028
029    public CommandHandler(String command, String description, ScoreboardStats plugin, String... aliases) {
030        this.plugin = plugin;
031        this.description = ChatColor.translateAlternateColorCodes('&', description);
032        this.permission = plugin.getName().toLowerCase() + ".command." + command;
033        this.subCommand = command;
034
035        this.aliases = Arrays.asList(aliases);
036    }
037
038    /**
039     * Get the main sub command
040     *
041     * @return the main sub command
042     */
043    public String getSubCommand() {
044        return subCommand;
045    }
046
047    /**
048     * Get all aliases of this command. All strings in this list can invoke this
049     * command
050     *
051     * @return all aliases
052     */
053    public List<String> getAliases() {
054        return aliases;
055    }
056
057    /**
058     * Get the description of this command
059     *
060     * @return the description
061     */
062    public String getDescription() {
063        return description;
064    }
065
066    /**
067     * Get the permission of this command
068     *
069     * @return the permission
070     */
071    public String getPermission() {
072        return permission;
073    }
074
075    /**
076     * Check if the player has enough permissions to execute this command. It
077     * will send a no permission if he doesn't have it.
078     *
079     * @param sender the executor
080     * @return whether the sender has enough permissions
081     */
082    public boolean hasPermission(CommandSender sender) {
083        if (sender.hasPermission(permission)) {
084            return true;
085        }
086
087        sender.sendMessage(Lang.get("noPermission"));
088        return false;
089    }
090
091    /**
092     * Executes the given subcommand
093     *
094     * @param sender Source of the command
095     * @param subCommand Command which was executed
096     * @param args The arguments passed to the command, including final partial argument to be completed
097     */
098    public abstract void onCommand(CommandSender sender, String subCommand, String... args);
099
100    /**
101     * Requests a list of possible completions for a command argument.
102     *
103     * @param sender Source of the command
104     * @param subCommand Command which was executed
105     * @param args The arguments passed to the command, including final partial argument to be completed
106     *
107     * @return A List of possible completions for the final argument, or null to default to the command executor
108     */
109    public List<String> onTabComplete(CommandSender sender, String subCommand, String... args) {
110        return null;
111    }
112}
